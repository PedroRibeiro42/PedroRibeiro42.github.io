#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  // Um array exemplo 
  vector<int> v{33, 4, 28, 18, 15, 2, 8, 17, 42, 39};
  
  cout << " Antes de ordenar: ";
  for (int x : v)
    cout << x << " ";
  cout << endl;
  
  // Chamada ao sort padrao da linguagem C++
  sort(v.begin(), v.end());
  
  cout << "Depois de ordenar: ";
  for (int x : v)
    cout << x << " ";
  cout << endl;

  return 0;
}
